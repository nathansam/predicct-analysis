cox.summary <- function(fit) {
  cat("Cox model summary:")
  fit %>%
    finalfit::fit2df(condense = FALSE) %>%
    knitr::kable(
      col.names = c(
        "Variable",
        "HR",
        "Lower 95%",
        "Upper 95%",
        "P-value"
      ),
      digits = 4
    ) %>%
    print()

  cat("\\newline

Diagnostics: \\newline

")

  cat("\\newline

::: {.panel-tabset}

      ")

  cat("

###### Proportional hazards assumption test \\newline

  ")
  cox.zph(fit)$table %>%
    knitr::kable(
      col.names = c(
        "",
        "Chi-squared statistic",
        "DF",
        "P-value"
      ),
      digits = 4
    ) %>%
    print()

  cat("\\newline

###### DF betas \\newline

  ")

  print(ggcoxdiagnostics(fit, type = "dfbeta"))

  cat("\\newline

###### Martingale residuals \\newline

      ")

  print(ggcoxdiagnostics(fit, type = "martingale", linear.predictions = TRUE))

  cat("\\newline

:::

      ")
  return()
}

cox.summary <- function(fit) {
  cat("Cox model summary:")
  fit %>%
    finalfit::fit2df(condense = FALSE) %>%
    knitr::kable(
      col.names = c(
        "Variable",
        "HR",
        "Lower 95%",
        "Upper 95%",
        "P-value"
      ),
      digits = 4
    ) %>%
    print()

  cat("\\newline

Diagnostics: \\newline

")

  cat("\\newline

::: {.panel-tabset}

      ")

  cat("

###### Proportional hazards assumption test \\newline

  ")
  cox.zph(fit)$table %>%
    knitr::kable(
      col.names = c(
        "",
        "Chi-squared statistic",
        "DF",
        "P-value"
      ),
      digits = 4
    ) %>%
    print()

  cat("\\newline

###### DF betas \\newline

  ")

  print(ggcoxdiagnostics(fit, type = "dfbeta"))

  cat("\\newline

###### Martingale residuals \\newline

      ")

  print(ggcoxdiagnostics(fit, type = "martingale", linear.predictions = TRUE))

  cat("\\newline

:::

      ")
  return()
}
