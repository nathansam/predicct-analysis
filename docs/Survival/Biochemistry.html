cox.summary <- function(fit, data = NULL) {
  cat("Cox model summary:")
  fit %>%
    finalfit::fit2df(condense = FALSE) %>%
    knitr::kable(
      col.names = c(
        "Variable",
        "HR",
        "Lower 95%",
        "Upper 95%",
        "P-value"
      ),
      digits = 4
    ) %>%
    print()
  
  if (!is.null (data)) {
    selector <- colnames(fit$x) %in% colnames(data) # Find continuous vars
    sd.vec <- apply(as.matrix(fit$x[, selector]), 2, sd)
    
    adj.table <- fit %>%
      finalfit::fit2df(condense = FALSE)
    adj.table <- adj.table[selector, ]
    adj.table[, 2:4] <-  adj.table[, 2:4] ^ sd.vec
    
    cat("\\newline

HR per SD: \\newline

")
    
    adj.table  %>%
      knitr::kable(
        col.names = c(
          "Variable",
          "HR",
          "Lower 95%",
          "Upper 95%",
          "P-value"),
      digits = 4,
      row.names = FALSE) %>%
      print()
  }
  
  cat("\\newline
  
Diagnostics: \\newline
  
")
  
  cat('\\newline
  
::: {.panel-tabset} 

      ')
  
  cat("
  
###### Proportional hazards assumption test \\newline

  ")
  cox.zph(fit)$table %>%
    knitr::kable(
      col.names = c(
        "",
        "Chi-squared statistic",
        "DF",
        "P-value"
      ),
      digits = 4
    ) %>%
    print()
  
  cat("\\newline

###### DF betas \\newline

  ")
  
  print(ggcoxdiagnostics(fit, type = "dfbeta"))
  
  cat("\\newline 
  
###### Martingale residuals \\newline
      
      ")
  
  print(ggcoxdiagnostics(fit, type = "martingale", linear.predictions = TRUE))
  
  cat("\\newline
  
:::

      ")
  return()
}

cox.summary <- function(fit, data = NULL) {
  cat("Cox model summary:")
  fit %>%
    finalfit::fit2df(condense = FALSE) %>%
    knitr::kable(
      col.names = c(
        "Variable",
        "HR",
        "Lower 95%",
        "Upper 95%",
        "P-value"
      ),
      digits = 4
    ) %>%
    print()
  
  if (!is.null (data)) {
    selector <- colnames(fit$x) %in% colnames(data) # Find continuous vars
    sd.vec <- apply(as.matrix(fit$x[, selector]), 2, sd)
    
    adj.table <- fit %>%
      finalfit::fit2df(condense = FALSE)
    adj.table <- adj.table[selector, ]
    adj.table[, 2:4] <-  adj.table[, 2:4] ^ sd.vec
    
    cat("\\newline

HR per SD: \\newline

")
    
    adj.table  %>%
      knitr::kable(
        col.names = c(
          "Variable",
          "HR",
          "Lower 95%",
          "Upper 95%",
          "P-value"),
      digits = 4,
      row.names = FALSE) %>%
      print()
  }
  
  cat("\\newline
  
Diagnostics: \\newline
  
")
  
  cat('\\newline
  
::: {.panel-tabset} 

      ')
  
  cat("
  
###### Proportional hazards assumption test \\newline

  ")
  cox.zph(fit)$table %>%
    knitr::kable(
      col.names = c(
        "",
        "Chi-squared statistic",
        "DF",
        "P-value"
      ),
      digits = 4
    ) %>%
    print()
  
  cat("\\newline

###### DF betas \\newline

  ")
  
  print(ggcoxdiagnostics(fit, type = "dfbeta"))
  
  cat("\\newline 
  
###### Martingale residuals \\newline
      
      ")
  
  print(ggcoxdiagnostics(fit, type = "martingale", linear.predictions = TRUE))
  
  cat("\\newline
  
:::

      ")
  return()
}
